\section{FACILITATOR GUIDE}
%===============================================================================

\subsection{Before the Session}
\begin{enumerate}
\item Prepare testing environment (clear browser cache, test dashboard functionality)
\item Have consent forms ready
\item Set up recording equipment (if approved)
\item Prepare backup devices/browsers if needed
\item Review participant background information
\end{enumerate}

\subsection{During the Session}
\subsubsection{Introduction (2-3 minutes):}
\begin{itemize}[leftmargin=*]
\item Welcome and thank participant
\item Explain purpose and process
\item Emphasize ``think aloud'' protocol
\item Get consent for recording/observation
\end{itemize}

\subsubsection{Task Completion (15-18 minutes):}
\begin{itemize}[leftmargin=*]
\item Read each task exactly as written
\item Don't lead or hint at solutions
\item Note time to complete each task
\item Observe navigation patterns and pain points
\item Only assist after 2 minutes of struggle
\end{itemize}

\subsubsection{Questionnaires (5-7 minutes):}
\begin{itemize}[leftmargin=*]
\item Allow participant to complete at their own pace
\item Be available for clarification
\item Encourage detailed, honest feedback
\end{itemize}

\subsection{After the Session}
\begin{enumerate}
\item Thank participant for their time
\item Debrief briefly (if appropriate)
\item Complete facilitator notes immediately
\item Organize and analyze results
\item Identify patterns and priority issues
\end{enumerate}

\subsection{Scoring Guidelines}
\subsubsection{SUS Score Interpretation:}
\begin{itemize}[leftmargin=*]
\item 80-100: Excellent
\item 70-79: Good  
\item 60-69: OK
\item 50-59: Poor
\item Below 50: Awful
\end{itemize}

\subsubsection{Task Success Criteria:}
\begin{itemize}[leftmargin=*]
\item Complete: All requirements met without assistance
\item Partial: Some requirements met or minimal assistance needed
\item Failed: Requirements not met or significant assistance required
\end{itemize}

\subsubsection{Issue Severity:}
\begin{itemize}[leftmargin=*]
\item Critical: Prevents task completion
\item Major: Causes significant difficulty or errors
\item Minor: Causes inconvenience or slight confusion
\end{itemize}
